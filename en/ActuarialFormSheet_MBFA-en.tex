\newcommand{\isvalinlist}[2]{%
	\def\valtolook{#1}%
	\def\foundmatch{false}%
	\@for\item:=#2\do{%
		\ifthenelse{\equal{\item}{\valtolook}}{%
			\def\foundmatch{true}%
		}{}%
	}%
}
